    使用 libpng¶

    需要准备好 input.bin，是一个 2048*1080/8 B 的二进制文件（内容每 1 bit 为一个像素）。

    png.c:

    #include <png.h>
#include <stdio.h>
#include <stdlib.h>

// [parameters]

unsigned int input_width = 2048;
unsigned int input_height = 1080;
unsigned int png_width = 1920;
unsigned int png_height = 1080;
const char* input_file = "input.bin";
const char* output_file = "output.png";

int main() {
  // [read binary input]

  unsigned int input_len = input_width * input_height / 8;
  unsigned char* input = (unsigned char*)malloc(input_len);
  FILE* input_f = fopen(input_file, "rb");
  fread(input, 8, input_len, input_f);
  fclose(input_f);

  for (unsigned int i = 0; i < input_width * input_height / 8; i++) {
    input[i] = reverse_byte[input[i]];
  }

  png_bytep row_pointers[input_height];
  for (png_uint_32 k = 0; k < input_height; k++) {
    row_pointers[k] = input + k * input_width / 8;
  }

  // [png encoding]

  FILE* output_f = fopen(output_file, "wb");
  png_structp png_ptr =
      png_create_write_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
  png_infop info_ptr = png_create_info_struct(png_ptr);

  png_init_io(png_ptr, output_f);
  png_set_IHDR(png_ptr, info_ptr, png_width, png_height, 1, PNG_COLOR_TYPE_GRAY,
               PNG_INTERLACE_NONE, PNG_COMPRESSION_TYPE_BASE,
               PNG_FILTER_TYPE_BASE);
  png_write_info(png_ptr, info_ptr);

  png_write_image(png_ptr, row_pointers);
  png_write_end(png_ptr, info_ptr);
  png_destroy_write_struct(&png_ptr, &info_ptr);
  fclose(output_f);

  return 0;
}
